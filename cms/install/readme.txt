目录文件说明

install/config	配置样本目录，最终会根据用户安装时的输入替换内容，覆盖相应文件
images			安装程序所需图片资源
sql				系统需要的sql文件
index.php		安装程序入口
apps.php		定义App列表，格式如下

array(
	array(
		'app' => 'system',		//app英文名，必填
		'name' => '系统',		//app中文名, 必填
		'description' => '',	//app说明，选填
		'is_core'	=>	1,		//是否核心组件（必装的），选填
		'is_model'	=>	1,		//是否模型，选填
	),
);
